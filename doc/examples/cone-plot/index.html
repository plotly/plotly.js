  type: "cone",
  x: [1], y: [1], z: [1],
  u: [1], v: [1], w: [0]
}]

  }

    multiple-3d-cone 

    var data = [{
      type: "cone",
      x: [1, 2, 3],
      y: [1, 2, 3],
      z: [1, 2, 3],
      u: [1, 0, 0],
      v: [0, 3, 0],
      w: [0, 0, 2],
      sizemode: "absolute",
      sizeref: 2,
      anchor: "tip",
      colorbar: {
        x: 0,
        xanchor: "right",
        side: "left"
      }
}]

var layout = {
    scene: {
      domain: {x: [0, 1]},
      camera: {
        eye: {x: -1.57, y: 1.36, z: 0.58}
      }
    },
    width: 800
}

Plotly.newPlot('myDiv', data, layout)
